227  void solve(double fill_factor, double tol, double pp_tol, int max_iter = -1, double minres_tol = 1e-6, double shift = 0.0) {
228  // A full factorization is equivalent to a fill factor of n and tol of 0
229  if (solve_type == solver_type::FULL) {
230  tol = 0.0;
231  fill_factor = A.n_cols();
232  }
233 
234  perm.reserve(A.n_cols());
235  cout << std::fixed << std::setprecision(3);
236 
237  double dif, total = 0;
238  clock_t start;
239 
240  if (equil_type == equilibration_type::BUNCH) {
241  start = clock();
242  A.sym_equil();
243  dif = clock() - start; total += dif;
244  printf(" Equilibration:\t\t%.3f seconds.\n", dif/CLOCKS_PER_SEC);
245  }
246 
247  if (reorder_type != reordering_type::NONE) {
248  start = clock();
249  std::string perm_name;
250  switch (reorder_type) {
251  case reordering_type::AMD:
252  A.sym_amd(perm);
253  perm_name = "AMD";
254  break;
255  case reordering_type::RCM:
256  A.sym_rcm(perm);
257  perm_name = "RCM";
258  break;
259  }
260 
261  dif = clock() - start; total += dif;
262  printf(" %s:\t\t\t\t%.3f seconds.\n", perm_name.c_str(), dif/CLOCKS_PER_SEC);
263 
264  start = clock();
265  A.sym_perm(perm);
266  dif = clock() - start; total += dif;
267  printf(" Permutation:\t\t\t%.3f seconds.\n", dif/CLOCKS_PER_SEC);
268  } else {
269  // no permutation specified, store identity permutation instead.
270  for (int i = 0; i < A.n_cols(); i++) {
271  perm.push_back(i);
272  }
273  }
274 
275  start = clock();
276  if (perform_inplace) {
277  A.ildl_inplace(D, perm, fill_factor, tol, pp_tol, piv_type);
278  } else {
279  A.ildl(L, D, perm, fill_factor, tol, pp_tol, piv_type);
280  }
281  dif = clock() - start; total += dif;
282 
283  std::string pivot_name;
284  if (piv_type == pivot_type::BKP) {
285  pivot_name = "BK";
286  } else if (piv_type == pivot_type::ROOK) {
287  pivot_name = "Rook";
288  }
289 
290  printf(" Factorization (%s pivoting):\t%.3f seconds.\n", pivot_name.c_str(), dif/CLOCKS_PER_SEC);
291  printf("Total time:\t\t\t%.3f seconds.\n", total/CLOCKS_PER_SEC);
292  if (perform_inplace) {
293  printf("L is %d by %d with %d non-zeros.\n", A.n_rows(), A.n_cols(), A.nnz() );
294  } else {
295  printf("L is %d by %d with %d non-zeros.\n", L.n_rows(), L.n_cols(), L.nnz() );
296  }
297  printf("\n");
298  fflush(stdout);
299 
300  // if there is a right hand side, it means the user wants a solve.
301  // TODO: refactor this solve to be in its own method, and separate
302  // factoring/minres solve phase
303  if (has_rhs) {
304  if (perform_inplace) {
305  printf("Inplace factorization cannot be used with the solver. Please try again without -inplace.\n");
306  } else {
307  // start timer in case we're doing a full solve
308  start = clock();
309 
310  // we've permuted and equilibrated the matrix, so we gotta apply
311  // the same permutation and equilibration to the right hand side.
312  // i.e. rhs = P'S*rhs
313  // 0. apply S
314  for (int i = 0; i < A.n_cols(); i++) {
315  rhs[i] = A.S[i]*rhs[i];
316  }
317 
318  // 1. apply P' (takes rhs[perm[i]] to rhs[i], i.e. inverse of perm,
319  // where perm takes i to perm[i])
320  vector<el_type> tmp(A.n_cols());
321  for (int i = 0; i < A.n_cols(); i++) {
322  tmp[i] = rhs[perm[i]];
323  }
324  rhs = tmp;
325 
326  if (solve_type == solver_type::FULL) {
327  printf("Solving matrix with direct solver...\n");
328  sol_vec.resize(A.n_cols(), 0);
329  // MINRES uses the preconditioned solver that
330  // splits the block D into |D|^(1/2).
331  // For the full solver we'll just solve D directly.
332  L.backsolve(rhs, sol_vec);
333  D.solve(sol_vec, tmp);
334  L.forwardsolve(tmp, sol_vec);
335  } else {
336  start = clock();
337 
338  if (solve_type == solver_type::MINRES) {
339  // finally, since we're preconditioning with M = L|D|^(1/2), we have
340  // to multiply M^(-1) to the rhs and solve the system
341  // M^(-1) * B * M'^(-1) y = M^(-1)P'*S*b
342  L.backsolve(rhs, tmp);
343  D.sqrt_solve(tmp, rhs, false);
344 
345  printf("Solving matrix with MINRES...\n");
346  // solve the equilibrated, preconditioned, and permuted linear system
347  minres(max_iter, minres_tol, shift);
348 
349  // now we've solved M^(-1)*B*M'^(-1)y = M^(-1)P'*S*b
350  // where B = P'SASPy.
351 
352  // but the actual solution is y = M' * P'S^(-1)*x
353  // so x = S*P*M'^(-1)*y
354 
355  // 0. apply M'^(-1)
356  D.sqrt_solve(sol_vec, tmp, true);
357  L.forwardsolve(tmp, sol_vec);
358  } else if (solve_type == solver_type::SQMR) {
359  printf("Solving matrix with SQMR...\n");
360  sqmr(max_iter, minres_tol);
361  }
362  }
363 
364  // 1. apply P
365  for (int i = 0; i < A.n_cols(); i++) {
366  tmp[perm[i]] = sol_vec[i];
367  }
368  sol_vec = tmp;
369 
370  // 2. apply S
371  for (int i = 0; i < A.n_cols(); i++) {
372  sol_vec[i] = A.S[i]*sol_vec[i];
373  }
374  dif = clock() - start;
375  printf("Solve time:\t%.3f seconds.\n", dif/CLOCKS_PER_SEC);
376  printf("\n");
377 
378  // save results
379  // TODO: refactor this to be in its own method
380  printf("Solution saved to output_matrices/outsol.mtx.\n");
381  save_vector(sol_vec, "output_matrices/outsol.mtx");
382  }
383  }
