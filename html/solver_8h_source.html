184  void solve(double fill_factor, double tol, double pp_tol, int max_iter = -1, double minres_tol = 1e-6, double shift = 0.0) {
185  // A full factorization is equivalent to a fill factor of n and tol of 0
186  if (full_solve) {
187  tol = 0.0;
188  fill_factor = A.n_cols();
189  }
190 
191  perm.reserve(A.n_cols());
192  cout << std::fixed << std::setprecision(3);
193  //gettimeofday(&tim, NULL);
194  //double t0=tim.tv_sec+(tim.tv_usec/1e6);
195  clock_t start = clock(); double dif, total = 0;
196 
197  if (equil == 1) {
198  A.sym_equil();
199  dif = clock() - start; total += dif;
200  printf(" Equilibration:\t\t%.3f seconds.\n", dif/CLOCKS_PER_SEC);
201  }
202 
203  if (reorder_scheme != 2) {
204  start = clock();
205  std::string perm_name;
206  switch (reorder_scheme) {
207  case 0:
208  A.sym_amd(perm);
209  perm_name = "AMD";
210  break;
211  case 1:
212  A.sym_rcm(perm);
213  perm_name = "RCM";
214  break;
215  }
216 
217  dif = clock() - start; total += dif;
218  printf(" %s:\t\t\t\t%.3f seconds.\n", perm_name.c_str(), dif/CLOCKS_PER_SEC);
219 
220  start = clock();
221  A.sym_perm(perm);
222  dif = clock() - start; total += dif;
223  printf(" Permutation:\t\t\t%.3f seconds.\n", dif/CLOCKS_PER_SEC);
224  } else {
225  // no permutation specified, store identity permutation instead.
226  for (int i = 0; i < A.n_cols(); i++) {
227  perm.push_back(i);
228  }
229  }
230 
231  start = clock();
232  if (perform_inplace) {
233  A.ildl_inplace(D, perm, fill_factor, tol, pp_tol, piv_type);
234  } else {
235  A.ildl(L, D, perm, fill_factor, tol, pp_tol, piv_type);
236  }
237  dif = clock() - start; total += dif;
238 
239  std::string pivot_name;
240  if (piv_type == pivot_type::BKP) {
241  pivot_name = "BK";
242  } else if (piv_type == pivot_type::ROOK) {
243  pivot_name = "Rook";
244  }
245 
246  printf(" Factorization (%s pivoting):\t%.3f seconds.\n", pivot_name.c_str(), dif/CLOCKS_PER_SEC);
247  printf("Total time:\t\t\t%.3f seconds.\n", total/CLOCKS_PER_SEC);
248  if (perform_inplace) {
249  printf("L is %d by %d with %d non-zeros.\n", A.n_rows(), A.n_cols(), A.nnz() );
250  } else {
251  printf("L is %d by %d with %d non-zeros.\n", L.n_rows(), L.n_cols(), L.nnz() );
252  }
253  printf("\n");
254  fflush(stdout);
255 
256  // if there is a right hand side, it means the user wants a solve.
257  // TODO: refactor this solve to be in its own method, and separate
258  // factoring/minres solve phase
259  if (has_rhs) {
260  if (perform_inplace) {
261  printf("Inplace factorization cannot be used with the solver. Please try again without -inplace.\n");
262  } else {
263  // start timer in case we're doing a full solve
264  start = clock();
265 
266  // we've permuted and equilibrated the matrix, so we gotta apply
267  // the same permutation and equilibration to the right hand side.
268  // i.e. rhs = P'S*rhs
269  // 0. apply S
270  for (int i = 0; i < A.n_cols(); i++) {
271  rhs[i] = A.S[i]*rhs[i];
272  }
273 
274  // 1. apply P' (takes rhs[perm[i]] to rhs[i], i.e. inverse of perm,
275  // where perm takes i to perm[i])
276  vector<el_type> tmp(A.n_cols());
277  for (int i = 0; i < A.n_cols(); i++) {
278  tmp[i] = rhs[perm[i]];
279  }
280  rhs = tmp;
281 
282  if (full_solve) {
283  printf("Solving matrix with direct solver...\n");
284  sol_vec.resize(A.n_cols(), 0);
285  // MINRES uses the preconditioned solver that
286  // splits the block D into |D|^(1/2).
287  // For the full solver we'll just solve D directly.
288  L.backsolve(rhs, sol_vec);
289  D.solve(sol_vec, tmp);
290  L.forwardsolve(tmp, sol_vec);
291  } else {
292  // finally, since we're preconditioning with M = L|D|^(1/2), we have
293  // to multiply M^(-1) to the rhs and solve the system
294  // M^(-1) * B * M'^(-1) y = M^(-1)P'*S*b
295  L.backsolve(rhs, tmp);
296  D.sqrt_solve(tmp, rhs, false);
297 
298  printf("Solving matrix with MINRES...\n");
299  start = clock();
300  // solve the equilibrated, preconditioned, and permuted linear system
301  minres(max_iter, minres_tol, shift);
302 
303  // now we've solved M^(-1)*B*M'^(-1)y = M^(-1)P'*S*b
304  // where B = P'SASPy.
305 
306  // but the actual solution is y = M' * P'S^(-1)*x
307  // so x = S*P*M'^(-1)*y
308 
309  // 0. apply M'^(-1)
310  D.sqrt_solve(sol_vec, tmp, true);
311  L.forwardsolve(tmp, sol_vec);
312  }
313 
314 
315 
316  // 1. apply P
317  for (int i = 0; i < A.n_cols(); i++) {
318  tmp[perm[i]] = sol_vec[i];
319  }
320  sol_vec = tmp;
321 
322  // 2. apply S
323  for (int i = 0; i < A.n_cols(); i++) {
324  sol_vec[i] = A.S[i]*sol_vec[i];
325  }
326  dif = clock() - start;
327  printf("Solve time:\t%.3f seconds.\n", dif/CLOCKS_PER_SEC);
328  printf("\n");
329 
330  // save results
331  // TODO: refactor this to be in its own method
332  printf("Solution saved to output_matrices/outsol.mtx.\n");
333  save_vector(sol_vec, "output_matrices/outsol.mtx");
334  }
335  }
