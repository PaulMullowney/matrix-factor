                if (block_size(i) == 2) {
                        out << i+2 << " " << i+1 << " " << off_diag.find(i)->second << "\n";
                        out << i+2 << " " << i+2 << " " << main_diag[i+1] << "\n";
                        i++;
                }
        }
        
        out.close();
        return true;
}
